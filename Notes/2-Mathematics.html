double pow_recursive(double a, int n)
{
    if (n == 0)
        return 1;
    if (n == 1)
        return a;
    double t = pow(a, n / 2);
    return t * t * pow(a, n % 2);
}

double pow_non_recursive(double a, int n)
{
    double ret = 1;
    while (n)
    {
        if (n % 2 == 1)
            ret *= a;
        
        a *= a;
        n >>= 1;
    }
    return ret;
}

Theoretically, both modes should perform the same in O(log(n))O(log(n))O(log(n)), but the recursive way perform slower as it occupies more resources.
